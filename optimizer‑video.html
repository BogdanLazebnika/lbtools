Повністю локальний відео-компресор













async function compress() { const file = document.getElementById('upload').files[0] if (!file) return alert("Завантаж відео") document.getElementById("status").innerText = "Завантаження FFmpeg..." if (!ffmpeg.isLoaded()) { await ffmpeg.load() } document.getElementById("status").innerText = "Обробка..." ffmpeg.FS('writeFile', 'input', await fetchFile(file)) const width = document.getElementById("width").value const fps = document.getElementById("fps").value const bitrate = document.getElementById("bitrate").value const format = document.getElementById("format").value const mute = document.getElementById("mute").checked let command = ['-i', 'input'] if (width) command.push('-vf', `scale=${width}:-1`) if (fps) command.push('-r', fps) if (bitrate) command.push('-b:v', bitrate + 'k') if (mute) command.push('-an') command.push('output.' + format) await ffmpeg.run(...command) const data = ffmpeg.FS('readFile', 'output.' + format) const blob = new Blob([data.buffer], { type: 'video/' + format }) const url = URL.createObjectURL(blob) const a = document.createElement('a') a.href = url a.download = 'compressed.' + format a.click() document.getElementById("status").innerText = "Готово!" }
